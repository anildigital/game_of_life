  def next_state
    cell = self

    cloned_cell = cell.clone # Use cloned cell as it would be next state of cell

    # kill the cell due to lonelyness
    cloned_cell.is_alive = false if cell.is_alive && cell.number_of_alive_neighbours < 2

    # kill the cell due overcrowdedness
    cloned_cell.is_alive = false if cell.is_alive && cell.number_of_alive_neighbours > 3

    # give birth to new cell when there are exactly 3 alive neighbours and current state is dead
    cloned_cell.is_alive = true if !(cell.is_alive) && cell.number_of_alive_neighbours == 3


    cloned_cell
  end
